#include <string>
#include<conio.h>
#include<cstring>


using namespace std; 

int main()
{
	setlocale(LC_ALL, "rus");

	char stroka1[50] = ""; //цифры в массиве значит сколько макс. символов можно записать.
	char stroka2[50] = ""; //цифры в массиве значит сколько макс. символов можно записать.
	cout << "Введите 2 строки, затем моя програма обьеденит их!" << endl;

	cout << "Вводите 1 строку: " << endl;
	cin.getline(stroka1, 50); // use getline, что бы юзер мог вводить целыми строками.
	cout << "Введите 2 строку: " << endl;
	cin.getline(stroka2, 50); // use getline, что бы юзер мог вводить целыми строками.

	cout << "Первая строка: " << stroka1 << endl;
	cout << "Вторая строка: " << stroka2 << endl;
	
	strcat_s(stroka1, stroka2); //функция stract(,) обьеденяет строки. В итоге строка 2 будем добавлена в строку 1.

	cout << "Обьеденяем и получается: " << stroka1 << endl;
	




	_getch();
	return 0;
}
