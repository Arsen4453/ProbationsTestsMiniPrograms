#include<iostream>
#include<conio.h>
using namespace std;
int x, y, z;
void xy(int y, int x)
{
	for (int i = 0; i < y; i++)
		cout << endl;
	for (int j = 0; j < x; j++)
		cout << " ";
}
int main()
{
	cout << "risuem domik";
	cout << "\nvedite koordinat y = "; cin >> y; 
	cout << "\nvedite koordinat x = "; cin >> x;
	cout << "\nvedite razmer z = "; cin >> z;
	for (int k = 0; k < (z + 1); k++) 
	{
		if (k == 0) 
		{
			xy(y, x); cout << " * ";
		}
		else if (k > 0 && k < z)
		{
			xy(1, (x - k)); cout << " / ";
			xy(0, k + k); cout << "\\";
		}
		else
		{
			xy(1, x - z - 1);
			for (int l = 0; l < (z); l++)
			{
				xy(0, 1); cout << "—";
			}
		}
	}
	for (int k = 0; k < (z + 1); k++) // рисование стен
	{
		xy(1, (x - z)); cout << " | ";
		xy(0, z + z); cout << " | ";
	}
	xy(1, x - z - 1);
	for (int l = 0; l < (z); l++)
	{
		xy(0, 1); cout << "-"; // рисование пол
	}
	cout << endl;
	_getch();

}
